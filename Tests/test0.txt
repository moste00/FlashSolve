class TFoo;
	rand bit [3:0] x;
	rand bit [3:0] y;
	rand bit [3:0] z;
	constraint C1 {
		unique {x,y,z};
	}
endclass
module tb;
    TFoo cls = new ();
    initial begin
        for (int i = 0; i < 1000; i++) begin
            cls.randomize();
            $display("i = %0d, x = %0d, y = %0d, z = %0d", i, cls.x, cls.y, cls.z);
        end
    end
endmodule
